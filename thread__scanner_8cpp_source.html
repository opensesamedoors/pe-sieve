35DWORD WINAPI enum_stack_thread(LPVOID lpParam)
36{
37 t_stack_enum_params* args = static_cast<t_stack_enum_params*>(lpParam);
38 if (!args || !args->c || !args->ctx) {
39 return STATUS_INVALID_PARAMETER;
40 }
41 size_t fetched = 0;
42 bool in_shc = false;
43 const pesieve::thread_ctx& c = *(args->c);
44#ifdef _WIN64
45 if (c.is64b) {
46 STACKFRAME64 frame = { 0 };
47
48 frame.AddrPC.Offset = c.rip;
49 frame.AddrPC.Mode = AddrModeFlat;
50 frame.AddrStack.Offset = c.rsp;
51 frame.AddrStack.Mode = AddrModeFlat;
52 frame.AddrFrame.Offset = c.rbp;
53 frame.AddrFrame.Mode = AddrModeFlat;
54
55 while (StackWalk64(IMAGE_FILE_MACHINE_AMD64, args->hProcess, args->hThread, &frame, args->ctx, NULL, SymFunctionTableAccess64, SymGetModuleBase64, NULL)) {
56 //std::cout << "Next Frame start:" << std::hex << frame.AddrPC.Offset << "\n";
57 const ULONGLONG next_addr = frame.AddrPC.Offset;
58 args->stack_frame.push_back(next_addr);
59 fetched++;
60 }
61 }
62#endif
63 if (!c.is64b) {
64 STACKFRAME frame = { 0 };
65
66 frame.AddrPC.Offset = c.rip;
67 frame.AddrPC.Mode = AddrModeFlat;
68 frame.AddrStack.Offset = c.rsp;
69 frame.AddrStack.Mode = AddrModeFlat;
70 frame.AddrFrame.Offset = c.rbp;
71 frame.AddrFrame.Mode = AddrModeFlat;
72
73 while (StackWalk(IMAGE_FILE_MACHINE_I386, args->hProcess, args->hThread, &frame, args->ctx, NULL, SymFunctionTableAccess, SymGetModuleBase, NULL)) {
74 const ULONGLONG next_addr = frame.AddrPC.Offset;
75 args->stack_frame.push_back(next_addr);
76 fetched++;
77 }
78 }
79 if (fetched) {
80 args->is_ok = true;
81 return STATUS_SUCCESS;
82 }
83 return STATUS_UNSUCCESSFUL;

86size_t pesieve::ThreadScanner::enumStackFrames(IN HANDLE hProcess, IN HANDLE hThread, IN LPVOID ctx, IN OUT thread_ctx& c)
87{
88 // do it in a new thread to prevent stucking...
89 t_stack_enum_params args(hProcess, hThread, ctx, c);
90 const size_t max_wait = 1000;
91 {
92 HANDLE enumThread = CreateThread(
93 NULL, // default security attributes
94 0, // use default stack size
95 enum_stack_thread, // thread function name
96 &args, // argument to thread function
97 0, // use default creation flags
98 0); // returns the thread identifiee
99
100 if (enumThread) {
101 DWORD wait_result = WaitForSingleObject(enumThread, max_wait);
102 if (wait_result == WAIT_TIMEOUT) {
103 std::cerr << "[!] Cannot retrieve stack frame: timeout passed!\n";
104 TerminateThread(enumThread, 0);
105 CloseHandle(enumThread);
106 return 0;
107 }
108 CloseHandle(enumThread);
109 }
110 }
111
112 if (!args.is_ok) {
113 return 0;
114 }
115
116 // filter:
117 size_t cntr = 0;
118 bool has_shellcode = false;
119 c.is_managed = false;
120 std::vector<ULONGLONG>::iterator itr;
121 for (itr = args.stack_frame.begin(); itr != args.stack_frame.end(); ++itr) {
122 cntr++;
123 ULONGLONG next_addr = *itr;
124#ifdef _DEBUG
125 resolveAddr(next_addr);
126#endif
127 bool is_curr_shc = false;
128 const ScannedModule* mod = modulesInfo.findModuleContaining(next_addr);
129 const std::string mod_name = mod ? mod->getModName() : "";
130 if (mod_name.length() == 0) {
131 has_shellcode = is_curr_shc = true;
132#ifdef _DEBUG
133 std::cout << std::hex << next_addr << " <=== SHELLCODE\n";
134#endif
135 }
136 if (!has_shellcode || is_curr_shc) {
137 // store the last address, till the first called shellcode:
138 c.ret_addr = next_addr;
139 }
140 // check if the found shellcode is a .NET JIT:
141 if (mod_name == "clr.dll") {
142 c.is_managed = true;
143#ifdef _DEBUG
144 std::cout << std::hex << next_addr << " <--- .NET\n";
145#endif
146 if (has_shellcode) break;
147 }
148 }
149 return cntr;
150}

292// if extended info given, allow to filter out from the scan basing on the thread state and conditions
293bool should_scan(const util::thread_info& info)
294{
295 if (!info.is_extended) {
296 return true;
297 }
298 const KTHREAD_STATE state = (KTHREAD_STATE)info.ext.state;
299 if (state == Running || state == Ready) {
300 return true;
301 }
302 if (state == Terminated) {
303 return false;
304 }
305 if (state == Waiting) {
306 if (info.ext.start_addr == 0) {
307 return true;
308 }
309 if (info.ext.wait_reason == DelayExecution
310 || info.ext.wait_reason == Suspended
311 || info.ext.wait_reason == Executive // the thread is waiting got the scheduler
312 || info.ext.wait_reason == UserRequest // i.e. WaitForSingleObject/WaitForMultipleObjects
313 || info.ext.wait_reason == WrUserRequest // i.e. when the thread calls GetMessage
314 )
315 {
316 return true;
317 }
318 }
319 return false;
320}

348#ifndef _DEBUG

350 if (!should_scan(info)) {
351 CloseHandle(hThread); // close the opened thread
352 my_report->status = SCAN_NOT_SUSPICIOUS;

354 }
355#endif

358
359 DWORD exit_code = 0;

361 CloseHandle(hThread);
362

366 return my_report;
367 }
368#ifdef _DEBUG

370 if (exit_code != STILL_ACTIVE)
371 std::cout << " ExitCode: " << exit_code;
372
373 if (c.ret_addr != 0) {
374 std::cout << std::hex << " Ret: " << c.ret_addr;
375 }
376 std::cout << "\n";
377#endif
378
379 if (exit_code != STILL_ACTIVE) {
380 my_report->status = SCAN_NOT_SUSPICIOUS;
381 return my_report;
382 }
383#ifdef _DEBUG
384 // if compiled in a debug mode, make this check AFTER scan
385 // (so that we can see first what was skipped)
386 if (!should_scan(info)) {
387 my_report->status = SCAN_NOT_SUSPICIOUS;
388 return my_report;
389 }
390#endif
391 bool is_shc = isAddrInShellcode(c.rip);
392 if (is_shc) {
393 if (reportSuspiciousAddr(my_report, c.rip, c)) {
394 return my_report;
395 }
396 }
397 if ((c.ret_addr != 0) && (c.is_managed == false)) {
398 is_shc = isAddrInShellcode(c.ret_addr);
