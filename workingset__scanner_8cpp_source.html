101 }
102 }
103#endif
104

107 }

109 code = (codeP || codeS);
110 }
111 my_report->has_shellcode = code;
112
113 if ( (this->args.obfuscated != OBFUSC_NONE && obfuscated) || ((this->args.shellcode != SHELLC_NONE) && code) ){

128bool pesieve::WorkingSetScanner::isPotentiallyExecutable(MemPageData &memPage, const t_data_scan_mode &mode)
129{
130 if (mode == pesieve::PE_DATA_NO_SCAN) {
131 return false;
132 }
133
134 // check preconditions:
135 const bool is_managed = this->processReport.isManagedProcess();
136 if (mode == pesieve::PE_DATA_SCAN_NO_DEP
137 && this->pDetails.isDEP && !is_managed)
138 {
139 return false;
140 }
141 if (mode == pesieve::PE_DATA_SCAN_DOTNET
142 && !is_managed)
143 {
144 return false;
145 }
146 // preconditions are fulfilled, now check the access:
147 const bool is_page_readable = pesieve::util::is_readable(memPage.mapping_type, memPage.protection);
148 if (mode != pesieve::PE_DATA_SCAN_INACCESSIBLE_ONLY) {
149 if (is_page_readable) {
150 return true;
151 }
152 }
153 if ((mode >= pesieve::PE_DATA_SCAN_INACCESSIBLE) || (mode == pesieve::PE_DATA_SCAN_INACCESSIBLE_ONLY)) {
154 if (this->pDetails.isReflection && (memPage.protection & PAGE_NOACCESS)) {
155 return true;
156 }
157 }
158 return false;

161WorkingSetScanReport* pesieve::WorkingSetScanner::scanExecutableArea(MemPageData &_memPage)
162{
163 if (!_memPage.load()) {
164 return nullptr;
165 }
166 // check for PE artifacts (regardless if it has shellcode patterns):
167 if (!isScannedAsModule(_memPage)) {
168 ArtefactScanner artefactScanner(this->processHandle, this->pDetails, _memPage, this->processReport);
169 WorkingSetScanReport *my_report1 = artefactScanner.scanRemote();
170 if (my_report1) {
171 //pe artefacts found
172 return my_report1;
173 }
174 }
175 if ((this->args.shellcode == SHELLC_NONE) && (this->args.obfuscated == OBFUSC_NONE)) {
176 // not a PE file, and we are not interested in shellcode or obfuscated contents, so just finish it here
177 return nullptr;
178 }
179
180 //report about shellcode:
181 ULONGLONG region_start = _memPage.region_start;
182 const size_t region_size = size_t(_memPage.region_end - region_start);
183 WorkingSetScanReport* my_report = new WorkingSetScanReport((HMODULE)region_start, region_size, SCAN_NOT_SUSPICIOUS);
184 if (!my_report) {
185 return nullptr;
186 }
187
188 if (!checkAreaContent(_memPage, my_report)) { // check for shellcode patterns & stats
189 my_report->status = SCAN_ERROR;
190 }
191 if (my_report->status == SCAN_NOT_SUSPICIOUS) {
192 // do not keep reports for not suspicious areas
193 delete my_report;
194 return nullptr;
195 }
196 my_report->has_pe = isScannedAsModule(_memPage) && this->processReport.hasModule(_memPage.region_start);
197 return my_report;

264WorkingSetScanReport* pesieve::WorkingSetScanner::scanRemote()
265{
266 MemPageData memPage(this->processHandle, this->pDetails.isReflection, this->memRegion.base, 0);
267 memPage.is_listed_module = this->processReport.hasModule(this->memRegion.base);
268
269 if (!memPage.isInfoFilled() && !memPage.fillInfo()) {
270#ifdef _DEBUG
271 std::cout << "[!] Could not fill: " << std::hex << memPage.start_va << " to: " << memPage.region_end << "\n";
272#endif
273 return nullptr;
274 }
275 // sanity checks to make sure that we are scanning the same page that was previously collected:
276 if (memPage.alloc_base != this->memRegion.alloc_base) {
277#ifdef _DEBUG
278 std::cerr << "WARNING: Alloc Base mismatch: " << std::hex << memPage.alloc_base << " vs " << this->memRegion.alloc_base << std::endl;
279#endif
280 return nullptr;
281 }
282 if ((memPage.region_end - memPage.region_start) != this->memRegion.size) {
283#ifdef _DEBUG
284 std::cerr << "WARNING: Size mismatch: " << std::hex << (memPage.region_end - memPage.region_start) << " vs " << this->memRegion.size << std::endl;
285#endif
286 return nullptr;
287 }
288
289 // is the page executable?
290 const bool is_any_exec = isExecutable(memPage);
291 if (!is_any_exec) {
292 // probably not interesting
293 return nullptr;
294 }
295
296 if (memPage.mapping_type == MEM_MAPPED && memPage.isRealMapping()) {
297 //probably legit
298 return nullptr;
299 }
300
301 if (memPage.mapping_type == MEM_IMAGE) {
302 memPage.loadModuleName();
303 memPage.loadMappedName();
304 if (!isScannedAsModule(memPage)) {
305 scanImg(memPage);
306 }
307 const size_t region_size = (memPage.region_end) ? (memPage.region_end - memPage.region_start) : 0;
308 if (this->processReport.hasModuleContaining(memPage.region_start, region_size)) {
309 // the area was already scanned
310 return nullptr;
311 }
312 }
313#ifdef _DEBUG
314 std::cout << std::hex << memPage.start_va << ": Scanning executable area" << std::endl;
315#endif
316 WorkingSetScanReport* my_report = this->scanExecutableArea(memPage);
317 if (!my_report) {
318 return nullptr;
319 }
320 my_report->is_executable = true;
321 my_report->protection = memPage.protection;
322 my_report->mapping_type = memPage.mapping_type;
323 my_report->mapped_name = memPage.mapped_name;
324 return my_report;
